# ASK-097 参考答案

```python
import asyncio
from typing import Awaitable, TypeVar

T = TypeVar("T")


async def bounded_gather(coros: list[Awaitable[T]], limit: int = 5) -> list[T]:
    """限并发版 asyncio.gather；任意失败收集后整体 raise ExceptionGroup。"""
    sem = asyncio.Semaphore(limit)

    async def _run(c: Awaitable[T]) -> T:
        async with sem:
            return await c

    results = await asyncio.gather(
        *[_run(c) for c in coros],
        return_exceptions=True,
    )
    errors = [r for r in results if isinstance(r, BaseException)]
    if errors:
        raise ExceptionGroup("bounded_gather failures", errors)  # Py 3.11+
    return results  # type: ignore[return-value]


# Demo
async def fetch(i: int):
    await asyncio.sleep(0.1)
    if i == 3:
        raise ValueError(f"fail {i}")
    return i * 2


async def main():
    coros = [fetch(i) for i in range(10)]
    try:
        out = await bounded_gather(coros, limit=3)
        print(out)
    except ExceptionGroup as eg:
        print(f"got {len(eg.exceptions)} errors")
        # 还可以 catch 特定子类型：except* ValueError as eg: ...

asyncio.run(main())
```

## 关键点

| 点 | 说明 |
|---|---|
| `Semaphore(limit)` | 控制并发数；超过 limit 的 coro await `sem.acquire()` 等待 |
| `async with sem` | acquire/release 的语法糖 |
| `return_exceptions=True` | 防止 gather 在第一个错时取消其他任务（题目要求"不取消"） |
| `ExceptionGroup`（Py 3.11+） | 一次抛多个错；3.10 及以下用 `Exception("group", errors)` |
| 顺序保留 | `gather` 按输入顺序返回结果 |
