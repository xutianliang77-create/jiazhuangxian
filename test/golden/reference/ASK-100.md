# ASK-100 参考答案

```ts
interface Entry<V> {
  value: V;
  expireAt: number | null;  // 绝对时间戳；null = 永不过期
}

export class LRUCache<K, V> {
  private map = new Map<K, Entry<V>>();

  constructor(private maxSize: number) {
    if (maxSize <= 0) throw new Error("maxSize must be > 0");
  }

  private isExpired(e: Entry<V>): boolean {
    return e.expireAt !== null && e.expireAt <= Date.now();
  }

  get(k: K): V | undefined {
    const e = this.map.get(k);
    if (!e) return undefined;
    if (this.isExpired(e)) {
      this.map.delete(k);
      return undefined;
    }
    // 刷新到队尾（hot）
    this.map.delete(k);
    this.map.set(k, e);
    return e.value;
  }

  has(k: K): boolean {
    const e = this.map.get(k);
    if (!e) return false;
    if (this.isExpired(e)) {
      this.map.delete(k);
      return false;
    }
    return true;
  }

  set(k: K, v: V, ttlMs?: number): void {
    if (this.map.has(k)) this.map.delete(k);
    // 满了驱逐 cold (queue head)
    while (this.map.size >= this.maxSize) {
      const firstKey = this.map.keys().next().value;
      if (firstKey === undefined) break;
      this.map.delete(firstKey as K);
    }
    this.map.set(k, {
      value: v,
      expireAt: ttlMs != null ? Date.now() + ttlMs : null,
    });
  }

  delete(k: K): boolean {
    return this.map.delete(k);
  }

  clear(): void {
    this.map.clear();
  }

  get size(): number {
    return this.map.size;
  }
}
```

## Demo

```ts
const c = new LRUCache<string, number>(3);
c.set("a", 1);
c.set("b", 2);
c.set("c", 3);
c.get("a");       // 1，移到队尾：[b, c, a]
c.set("d", 4);    // 满了，驱逐 b：[c, a, d]
c.has("b");       // false

const c2 = new LRUCache<string, string>(10);
c2.set("token", "secret", 100);  // 100ms TTL
setTimeout(() => console.log(c2.get("token")), 200);  // undefined
```

## 关键点

| 点 | 说明 |
|---|---|
| 利用 Map 插入顺序 | ECMAScript 规范保证 Map 按插入顺序迭代；第一个 key = 最 cold |
| `get` 中 delete + set | O(1) 把 key 移到队尾（hot） |
| `has` 不刷顺序 | 题目要求不更新冷热；但仍要 evict 过期 |
| TTL 用 absolute timestamp | 比相对 ttl 简单：`expireAt <= Date.now()` 即过期 |
| O(1) | Map.delete / set / keys().next() 都是 O(1) |

|---|---|
