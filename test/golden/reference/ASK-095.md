```python
def load_env(path: str) -> dict[str, str]:
    """读 .env 风格文件，返回 KEY -> VALUE dict。不依赖 python-dotenv。"""
    out: dict[str, str] = {}
    with open(path, "r", encoding="utf-8") as f:
        for raw in f:
            line = raw.strip()
            if not line or line.startswith("#"):
                continue
            if "=" not in line:
                continue
            k, _, v = line.partition("=")
            k = k.strip()
            v = v.strip()
            # 剥离两侧成对引号
            if len(v) >= 2 and v[0] == v[-1] and v[0] in ("'", '"'):
                v = v[1:-1]
            out[k] = v
    return out


# Demo
if __name__ == "__main__":
    import io, tempfile
    sample = '''
# Comment
FOO=bar
BAZ="hello world"
QUX='single quoted'
URL=postgres://u:p@localhost/db?sslmode=require
EMPTY=
'''
    with tempfile.NamedTemporaryFile("w", suffix=".env", delete=False) as f:
        f.write(sample); fname = f.name
    print(load_env(fname))
    # {'FOO': 'bar', 'BAZ': 'hello world', 'QUX': 'single quoted',
    #  'URL': 'postgres://u:p@localhost/db?sslmode=require', 'EMPTY': ''}
```

## 关键点

- `partition("=")`：value 内部含 `=`（如 URL）也不会被拆错
- 引号 strip：只在 `'...'` 或 `"..."` 成对时 strip
- 注释 / 空行跳过
- UTF-8 显式
