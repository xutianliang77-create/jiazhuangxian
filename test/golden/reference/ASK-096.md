# ASK-096 参考答案

```ts
export function deepMerge<T>(target: T, source: Partial<T>): T {
  const isPlainObj = (v: unknown): v is Record<string, unknown> =>
    typeof v === "object"
    && v !== null
    && !Array.isArray(v)
    && Object.getPrototypeOf(v) === Object.prototype;

  if (!isPlainObj(target) || !isPlainObj(source)) {
    return (source !== undefined ? source : target) as T;
  }

  const out: Record<string, unknown> = { ...target };
  for (const k of Object.keys(source)) {
    const sv = (source as Record<string, unknown>)[k];
    const tv = out[k];
    out[k] = isPlainObj(tv) && isPlainObj(sv)
      ? deepMerge(tv, sv as never)
      : sv;
  }
  return out as T;
}
```

## Demo

```ts
const a = { x: 1, nested: { p: 'old', q: 1 }, arr: [1, 2] };
const b = { nested: { p: 'new', r: 3 }, arr: [9] };
console.log(deepMerge(a, b as Partial<typeof a>));
// { x: 1, nested: { p: 'new', q: 1, r: 3 }, arr: [9] }
console.log(a);  // 原对象不变
```

## 关键点

| 点 | 说明 |
|---|---|
| `isPlainObj` 检查 prototype | 排除 Date / Map / RegExp / 自定义 class，只对 `{}` 字面量做深合 |
| 数组直接覆盖 | 数组合并语义复杂（concat? union? by index?）—— 默认覆盖最不出错 |
| `{ ...target }` 复制 | 不修改原对象（pure function） |
| 递归 | 嵌套 object 才递归，叶子直接赋 |

## 不要这么做

```ts
// ❌ 用 Object.assign 是浅拷贝
return Object.assign({}, target, source);

// ❌ 用 lodash.merge：违题约束 + 引依赖

// ❌ JSON.parse(JSON.stringify) 深克隆：丢类型 / Date / undefined
```
