# ASK-098 参考答案

```ts
export function createDebouncedSearch<T>(
  fetcher: (q: string, signal: AbortSignal) => Promise<T>,
  ms: number,
) {
  let timer: ReturnType<typeof setTimeout> | null = null;
  let controller: AbortController | null = null;
  let pendingResolve: ((v: T) => void) | null = null;
  let pendingReject: ((e: unknown) => void) | null = null;

  const cancel = () => {
    if (timer) clearTimeout(timer);
    timer = null;
    controller?.abort();
    controller = null;
    if (pendingReject) pendingReject(new Error("cancelled"));
    pendingResolve = null;
    pendingReject = null;
  };

  const search = (q: string): Promise<T> => {
    cancel();
    return new Promise<T>((resolve, reject) => {
      pendingResolve = resolve;
      pendingReject = reject;
      timer = setTimeout(() => {
        controller = new AbortController();
        fetcher(q, controller.signal).then(
          (v) => pendingResolve?.(v),
          (e) => pendingReject?.(e),
        );
      }, ms);
    });
  };

  return { search, cancel };
}
```

## Demo

```ts
const fakeFetcher = (q: string, signal: AbortSignal) =>
  new Promise<string[]>((resolve, reject) => {
    const t = setTimeout(() => resolve([`result for ${q}`]), 500);
    signal.addEventListener("abort", () => {
      clearTimeout(t);
      reject(new Error("aborted"));
    });
  });

const { search, cancel } = createDebouncedSearch(fakeFetcher, 200);

search("a"); // 200ms 后会发；这次会被下一次 cancel
search("ab"); // 200ms 后会发；这次也被 cancel
search("abc"); // 200ms 后真发
//   .then(console.log)  → ["result for abc"]
```

## 关键点

| 点 | 说明 |
|---|---|
| `clearTimeout(timer)` | 防止旧的 setTimeout 触发 |
| `controller?.abort()` | 取消正在跑的 fetch（fetcher 监听 signal） |
| 旧 promise reject | 调用方 await 旧 search 时不会无限挂着 |
| 公开 `cancel()` | 调用方主动取消（如组件 unmount） |

## 反例

```ts
// 反例 1：只 clearTimeout 不 abort
clearTimeout(timer); // 但已发出的 fetch 不停 → 浪费带宽 + 可能后到的结果覆盖前面

// 反例 2：不 reject 旧 promise → 调用方 await 永远 pending（内存泄漏）

// 反例 3：用 lodash.debounce —— 题目禁，且不带 cancel + abort 一体
```
