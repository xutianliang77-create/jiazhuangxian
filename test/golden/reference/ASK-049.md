# ASK-049 参考答案

## 参考实现

```ts
export async function* parseSseStream(
  input: ReadableStream<Uint8Array>,
  signal?: AbortSignal,
): AsyncIterable<{ event: string; data: string }> {
  const reader = input.getReader();
  const decoder = new TextDecoder("utf-8");
  let buf = "";
  let curEvent = "message";
  const dataLines: string[] = [];

  const onAbort = () => {
    reader.cancel().catch(() => {});
  };
  signal?.addEventListener("abort", onAbort);

  try {
    while (true) {
      if (signal?.aborted) return;
      const { value, done } = await reader.read();
      if (done) break;
      buf += decoder.decode(value, { stream: true });

      let idx: number;
      while ((idx = buf.indexOf("\n")) !== -1) {
        const line = buf.slice(0, idx).replace(/\r$/, "");
        buf = buf.slice(idx + 1);

        if (line === "") {
          // 空行 = 消息边界
          if (dataLines.length > 0) {
            yield { event: curEvent, data: dataLines.join("\n") };
            dataLines.length = 0;
            curEvent = "message";
          }
        } else if (line.startsWith(":")) {
          // SSE comment，忽略
        } else if (line.startsWith("event:")) {
          curEvent = line.slice(6).trim();
        } else if (line.startsWith("data:")) {
          dataLines.push(line.slice(5).replace(/^ /, ""));
        }
        // id: / retry: 行此实现忽略
      }
    }
    // flush 剩余（流结束但没空行）
    if (dataLines.length > 0) {
      yield { event: curEvent, data: dataLines.join("\n") };
    }
  } finally {
    signal?.removeEventListener("abort", onAbort);
    reader.releaseLock();
  }
}
```

## Demo

```ts
import { parseSseStream } from "./parseSseStream.js";

const sseSample = [
  "event: chunk\n",
  "data: hello",
  " world\n",
  "\n",
  "data: line1\n",
  "data: line2\n",
  "\n",
].join("");

const stream = new ReadableStream({
  start(controller) {
    controller.enqueue(new TextEncoder().encode(sseSample));
    controller.close();
  },
});

for await (const ev of parseSseStream(stream)) {
  console.log(ev);
}
// → { event: 'chunk', data: 'hello world' }
//   { event: 'message', data: 'line1\nline2' }
```

## 关键点

| 要点 | 说明 |
|---|---|
| `async function*` | 实现 `AsyncIterable`，可被 `for await` 消费 |
| `TextDecoder({ stream: true })` | 处理 UTF-8 多字节 chunk 边界（一个汉字 3 字节，可能跨 chunk） |
| `buf` 累计 | partial line 拼接，等 `\n` 才切分 |
| `signal?.aborted` | 每次 read 前 check；abort 后立即 return |
| `reader.cancel()` | 被 abort 时主动取消上游，避免 socket 泄漏 |
| 空行边界 | SSE 协议规定连续空行分隔消息 |
| `\r\n` 兼容 | `replace(/\r$/, "")` 处理 Windows 风格行尾 |
